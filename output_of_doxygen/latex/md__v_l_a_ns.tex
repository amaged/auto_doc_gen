\#\#\# 
\begin{DoxyCode}{0}

\DoxyCodeLine{=========}
\DoxyCodeLine{}
\DoxyCodeLine{}
\DoxyCodeLine{---- Create an IP interface to a VLAN for routing or other applications}
\DoxyCodeLine{}
\DoxyCodeLine{switch\# net add vlan 100 ip address 10.0.0.1/24}
\DoxyCodeLine{switch\# net commit}
\DoxyCodeLine{}
\DoxyCodeLine{}
\DoxyCodeLine{---- Create a dual stack IP interface to a VLAN for routing or other applications}
\DoxyCodeLine{}
\DoxyCodeLine{switch\# net add vlan 100 ip address 10.0.0.1/24}
\DoxyCodeLine{switch\# net add vlan 100 ipv6 address fd00:2:0:1::1/64}
\DoxyCodeLine{switch\# net commit}
\DoxyCodeLine{}
\DoxyCodeLine{}
\DoxyCodeLine{---- Create a dual stack IP interface to a VLAN for routing or other applications}
\DoxyCodeLine{     along with a virtual interface for virtual routing redundancy.  The reserved}
\DoxyCodeLine{     range of MAC addresses for VRR is 00:00:5E:00:01:00 to 00:00:5E:00:01:ff.}
\DoxyCodeLine{}
\DoxyCodeLine{switch\# net add vlan 100 ip address 10.0.0.1/24}
\DoxyCodeLine{switch\# net add vlan 100 ipv6 address fd00:2:0:1::1/64}
\DoxyCodeLine{switch\# net add vlan 100 ip address-virtual 00:00:5e:00:01:01 10.0.0.254/24}
\DoxyCodeLine{switch\# net add vlan 100 ipv6 address-virtual 00:00:5e:00:01:01 fd00:2:0:1::254/64}
\DoxyCodeLine{switch\# net commit}
\end{DoxyCode}
